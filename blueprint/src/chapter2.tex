\section{Analysis of the Euler Method}

\noindent\textbf{Context.}
The theorem bounds the global error \(y(x)-\widehat{y}(x)\) for a numerical method applied to an ODE.  
Variables: \(f\) (ODE function, Lipschitz constant \(L\)), \(y(x)\) (exact solution), \(\widehat{y}(x)\) (numerical approximation), \(H\) (max step size), \(m\) (bound on local truncation error), \(x_{0}\) (initial point).  
Referenced equations:
\begin{align*}
(212a)&\quad \alpha(x)=\|y(x)-\widehat{y}(x)\|,\\
(212b)&\quad \beta(x)=\alpha'(x),\\
(212c)&\quad E(x)=\int_{x_{0}}^{x}\beta(\xi)d\xi,\\
(212d)&\quad |\beta(x)|\le L\alpha(x),\\
(212e)&\quad \alpha(x)=E(x)+\int_{x_{0}}^{x}\beta(\xi)d\xi.
\end{align*}

\begin{theorem}[Global truncation error (Euler) (212A)]

  \label{thm:212A}
  \lean{Butcher.Ch2.Thm212A}
  \uses{def:110A}
Assuming that \( f \) satisfies a Lipschitz condition, with constant \( L \), the global error satisfies the bound
\[
\| y(x) - \widehat{y}(x) \| \leq 
\begin{cases}
\| y(x_0) - \widehat{y}(x_0) \| + Hm(x - x_0), & L = 0, \\[6pt]
\exp((x - x_0)L) \| y(x_0) - \widehat{y}(x_0) \| + \dfrac{\exp((x - x_0)L) - 1}{L} Hm, & L > 0.
\end{cases}
\]
\end{theorem}

\noindent\textbf{Context.}
The theorem concerns the convergence of the Euler method for solving an ODE. A sequence of approximations \(y_n(x)\) is computed using the Euler method with stepsize bounded by \(H_n\) and initial error bounded by \(K_n\), where \(H_n\) and \(K_n\) tend to zero as \(n\) increases. The global error \(D_n\) is defined as the difference between the exact solution \(y(x)\) and the approximation \(y_n(x)\).

Variables:
\begin{itemize}
    \item \(y(x)\): exact solution of the differential equation
    \item \(y_n(x)\): approximation number \(n\) computed by the Euler method
    \item \(D_n\): global error for approximation \(n\), defined as \(y(x) - y_n(x)\)
    \item \(H_n\): greatest stepsize in approximation \(n\)
    \item \(K_n\): bound on the initial error \(y(x_0) - y_n(x_0)\) for approximation \(n\)
    \item \(f\): function defining the ODE, assumed to satisfy a Lipschitz condition
\end{itemize}

\begin{theorem}[Convergence of the Euler method (213A)]

  \label{thm:213A}
  \lean{Butcher.Ch2.Thm213A}
  \uses{thm:212A}
Under the conditions stated in the above discussion, $D_n \to 0$ as $n \to \infty$.
\end{theorem}

\begin{proof}
  \proves{thm:213A}
  \uses{thm:212A}
This result follows immediately from the bound on accumulated errors given by Theorem~\ref{thm:212A}.

The property expressed in this theorem is known as `convergence'. In searching for other numerical methods that are suitable for solving initial value problems, attention is usually limited to convergent methods. The reason for this is clear: a non-convergent method is likely to give increasingly meaningless results as greater computational effort is expended through the use of smaller stepsizes.

Because the bound used in the proof of Theorem~\ref{thm:213A} holds not only for $x = \overline{x}$, but also for all $x \in [x_0, \overline{x}]$, we can state a uniform version of this result.
\end{proof}

\noindent\textbf{Context.}
The theorem concerns the convergence of the Euler method for solving an initial value problem. A sequence of approximations \(y_n\) is constructed using the Euler method with stepsizes bounded by \(H_n\) and initial errors bounded by \(K_n\), both tending to zero as \(n \to \infty\). The function \(f\) satisfies a Lipschitz condition.

Variables:
\begin{itemize}
    \item \(y\): the exact solution to the initial value problem.
    \item \(y_n\): approximation \(n\), computed using the Euler method starting from an approximation to \(y(x_0)\) with a sequence of positive steps.
    \item \(x_0\): initial \(x\)-value for the ODE.
    \item \(x\): the final \(x\)-value (or a generic point in the interval).
    \item \(f\): the function in the ODE \(y' = f(x, y)\), assumed to satisfy a Lipschitz condition.
    \item \(H_n\): an upper bound on the greatest stepsize for approximation \(n\), with \(H_n \to 0\) as \(n \to \infty\).
    \item \(K_n\): a bound on the initial error \(\|y(x_0) - y_n(x_0)\|\), with \(K_n \to 0\) as \(n \to \infty\).
\end{itemize}

\begin{theorem}[Euler method uniform convergence theorem (213B)]

  \label{thm:213B}
  \lean{Butcher.Ch2.Thm213B}
  \uses{thm:213A}
Under the conditions of Theorem 213A,
\[
\sup_{x\in[x_0,\overline{x}]} \bigl| y(x) - y_n(x) \bigr| \to 0
\]
as $n \to \infty$.
\end{theorem}
